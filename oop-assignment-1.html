2. HashMap Implementation.

Program : 
    
        import java.util.* ;

        public class HashMapMethods
        {
        static void println(Object o){
        System.out.println(o);
        }

        public static void main(String[] args) {

        HashMap hm = new HashMap();
        # put key and value into hashMap
        hm.put("1","One");
        hm.put("2","Two");
        hm.put("3","Three");
        hm.put("4","Six");
        hm.put("5","Five");

        println("HashMap : "+hm);
        #size of HashMap
        println("Size : "+hm.size());
        #check whether HashMap contains the given key
        println("Contains Key ('1') : "+hm.containsKey("1"));
        #check whether HashMap contains the given value
        println("Contains Value ('Two') : "+hm.containsValue("Two"));
        # get the value associated with given key 
        println("Get ('3') : "+hm.get("3"));
        # replace the value for the specified key
        println("Replace ('4','Four') : "+hm.replace("4","Four"));
        println("HashMap : "+hm);
        # add key and value to the HashMap if it is absent
        hm.putIfAbsent("0","Zero");

        println("Put If Absent ('0','Zero') : "+hm);
        # get a copy the hashMap
        Object hm_clone = hm.clone();

        println("Clone : "+hm_clone);

        }
        }

Output : 
    
        HashMap : {1=One, 2=Two, 3=Three, 4=Six, 5=Five}
        Size : 5
        Contains Key ('1') : true
        Contains Value ('Two') : true
        Get ('3') : Three
        Replace ('4','Four') : Six
        HashMap : {1=One, 2=Two, 3=Three, 4=Four, 5=Five}
        Put If Absent ('0','Zero') : {0=Zero, 1=One, 2=Two, 3=Three, 4=Four, 5=Five}
        Clone : {0=Zero, 1=One, 2=Two, 3=Three, 4=Four, 5=Five}
3. HashSet Implementation.

Program : 
    
        import java.util.* ;

        public class HashSetMethods
        {
        static void println(Object o){
        System.out.println(o);
        }

        public static void main(String[] args) {

        HashSet hs = new HashSet();
        # Add items to HashSet
        hs.add("One");
        hs.add("Two");
        hs.add("Three");
        hs.add("Four");
        hs.add("Five");

        println("HashSet : "+hs);
        # get size of HashSet
        println("Size : "+hs.size());
        # check whether HashSet contains the specified item
        println("Contains ('Two') : "+hs.contains("Two"));
        # check whether HashSet is empty
        println("Is Empty : "+hs.isEmpty());
        # remove the specified value from hashSet
        hs.remove("Five");
        println("Remove ('Five') : "+hs);

        println("Iterator values : ");
        # Iterate the HashSet
        Iterator it = hs.iterator();
            # get a cloned HashSet
            Object cloned = hs.clone();

            while(it.hasNext()){
            println(it.next());
            }
            # clear the HashSet
            hs.clear();

            println("Clear Set : "+hs);

            println("Cloned : "+cloned);

            }
            }

        HashSet : [Five, One, Four, Two, Three]

        import java.util.* ;

        public class TreeMapMethods
        {
        static void println(Object o){
        System.out.println(o);
        }

        public static void main(String[] args) {

        TreeMap tm = new TreeMap();
                # add the key, value pair to the TreeMap 
                tm.put("1","One");
                tm.put("2","Two");
                tm.put("3","Three");
                tm.put("4","Four");
                tm.put("5","Five");

                println("TreeMap : "+tm);
                # get the Size of the TreeMap
                println("Size : "+tm.size());
                # get the first occuring value for key 
                println("Ceiling Key ('0') : "+tm.ceilingKey("0"));
                # get the first occuring pair
                println("Ceiling Entry('0') : "+tm.ceilingEntry("0"));
                # get the first pair in the TreeMap
                println("First Entry : "+tm.firstEntry());
                # get the last pair in the TreeMap
                println("Last Entry : "+tm.lastEntry());

                println("HashCode : "+tm.hashCode());
                # get the first key from the TreeMap
                println("FirstKey : "+tm.firstKey());
                # get the last key from the TreeMap
                println("LasttKey : "+tm.lastKey());

                println("Contains Key ('1') : "+tm.containsKey("1"));
                println("Contains Value ('One') : "+tm.containsValue("One"));

                }
                }

Output :
